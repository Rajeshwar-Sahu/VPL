// VPL Tournament — Google Apps Script Backend v7
// DEPLOY: Web App | Execute as: Me | Who has access: Anyone
// ============================================================
const SHEET_NAME = 'vpl_data';
const PROP_PW = 'VPL_ADMIN_PW';
const DEFAULT_PW = 'vpl2024';

// ── CORS helper — every response must include CORS headers ──
function cors(output) {
  return output
    .setMimeType(ContentService.MimeType.JSON);
  // Note: GAS automatically adds CORS headers when deployed as
  // "Anyone" web app. No manual header setting needed.
}

function json(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

// ── GET handler: pull data or check password ──
function doGet(e) {
  try {
    const action = (e.parameter && e.parameter.action) ? e.parameter.action : 'pull';
    if (action === 'pull') {
      return json(getData());
    }
    if (action === 'checkpw') {
      const stored = PropertiesService.getScriptProperties().getProperty(PROP_PW) || DEFAULT_PW;
      return json({ ok: e.parameter.pw === stored });
    }
    return json({ error: 'Unknown GET action: ' + action });
  } catch(err) {
    return json({ error: 'doGet error: ' + err.message });
  }
}

// ── POST handler: all writes come through here ──
function doPost(e) {
  try {
    if (!e.postData || !e.postData.contents) {
      return json({ error: 'No POST body received' });
    }
    const b = JSON.parse(e.postData.contents);
    const action = b.action || '';
    const stored = PropertiesService.getScriptProperties().getProperty(PROP_PW) || DEFAULT_PW;

    // Public actions — no password required
    if (action === 'fanvote') {
      const d = getData();
      const m = d.matches.find(function(x){ return x.id === b.matchId; });
      if (!m) return json({ error: 'Match not found' });
      m.fanVoteHome = (m.fanVoteHome || 0) + (b.side === 'home' ? 1 : 0);
      m.fanVoteAway = (m.fanVoteAway || 0) + (b.side === 'away' ? 1 : 0);
      saveData(d);
      return json({ ok: true });
    }

    if (action === 'photovote') {
      const d = getData();
      const p = d.photos.find(function(x){ return String(x.id) === String(b.photoId); });
      if (!p) return json({ error: 'Photo not found' });
      p.votes = (p.votes || 0) + 1;
      saveData(d);
      return json({ ok: true });
    }

    if (action === 'uploadphoto') {
      if (!b.photo) return json({ error: 'No photo data' });
      const d = getData();
      d.photos.push(b.photo);
      saveData(d);
      return json({ ok: true });
    }

    // Admin actions — password required
    if (b.pw !== stored) {
      return json({ error: 'Unauthorized — wrong password' });
    }

    if (action === 'changepw') {
      if (!b.newpw || b.newpw.length < 4) return json({ error: 'Password too short' });
      PropertiesService.getScriptProperties().setProperty(PROP_PW, b.newpw);
      return json({ ok: true });
    }

    const d = getData();

    if (action === 'addTeam') {
      if (!b.team) return json({ error: 'No team data' });
      // Prevent duplicates
      const exists = d.teams.some(function(t){ return t.name.toLowerCase() === b.team.name.toLowerCase(); });
      if (exists) return json({ error: 'Team name already exists' });
      d.teams.push(b.team);
      if (b.team.id >= d.nextTeamId) d.nextTeamId = b.team.id + 1;
      saveData(d);
      return json({ ok: true, state: d });
    }

    if (action === 'deleteTeam') {
      d.teams = d.teams.filter(function(t){ return t.id !== b.teamId; });
      saveData(d);
      return json({ ok: true, state: d });
    }

    if (action === 'editTeam') {
      const t = d.teams.find(function(t){ return t.id === b.teamId; });
      if (!t) return json({ error: 'Team not found' });
      t.name = b.name; t.pool = b.pool;
      saveData(d);
      return json({ ok: true, state: d });
    }

    if (action === 'setTeamLogo') {
      const t = d.teams.find(function(t){ return t.id === b.teamId; });
      if (!t) return json({ error: 'Team not found' });
      t.logo = b.logo;
      saveData(d);
      return json({ ok: true, state: d });
    }

    if (action === 'addMatch') {
      if (!b.match) return json({ error: 'No match data' });
      d.matches.push(b.match);
      if (b.match.id >= d.nextMatchId) d.nextMatchId = b.match.id + 1;
      saveData(d);
      return json({ ok: true, state: d });
    }

    if (action === 'editMatch') {
      const m = d.matches.find(function(m){ return m.id === b.matchId; });
      if (!m) return json({ error: 'Match not found' });
      m.homeId = b.homeId; m.awayId = b.awayId;
      if (b.date) m.date = b.date;
      if (b.venue !== undefined) m.venue = b.venue;
      saveData(d);
      return json({ ok: true, state: d });
    }

    if (action === 'deleteMatch') {
      d.matches = d.matches.filter(function(m){ return m.id !== b.matchId; });
      saveData(d);
      return json({ ok: true, state: d });
    }

    if (action === 'enterScore') {
      const m = d.matches.find(function(m){ return m.id === b.matchId; });
      if (!m) return json({ error: 'Match not found' });
      m.homeScore = b.homeScore; m.awayScore = b.awayScore;
      m.homePts = b.homePts || 0; m.awayPts = b.awayPts || 0;
      m.status = 'completed';
      var total = (m.fanVoteHome || 0) + (m.fanVoteAway || 0);
      if (total === 0) total = 1;
      if (b.homeScore > b.awayScore) { m.fanVoteHome = total; m.fanVoteAway = 0; }
      else { m.fanVoteAway = total; m.fanVoteHome = 0; }
      saveData(d);
      return json({ ok: true, state: d });
    }

    if (action === 'addDomain') {
      if (!d.allowedDomains.includes(b.domain)) d.allowedDomains.push(b.domain);
      saveData(d);
      return json({ ok: true, state: d });
    }

    if (action === 'removeDomain') {
      d.allowedDomains = d.allowedDomains.filter(function(x){ return x !== b.domain; });
      saveData(d);
      return json({ ok: true, state: d });
    }

    if (action === 'setPlayoffFormat') {
      d.playoffFormat = b.format;
      saveData(d);
      return json({ ok: true, state: d });
    }

    return json({ error: 'Unknown action: ' + action });

  } catch(err) {
    return json({ error: 'doPost error: ' + err.message });
  }
}

// ── Sheet helpers ──
function getData() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName(SHEET_NAME);
  if (!sh) {
    sh = ss.insertSheet(SHEET_NAME);
    sh.getRange(1,1).setValue('json');
    sh.getRange(2,1).setValue(JSON.stringify(defaultState()));
  }
  var val = sh.getRange(2,1).getValue();
  if (!val || val === '') {
    return defaultState();
  }
  try {
    return JSON.parse(val);
  } catch(e) {
    return defaultState();
  }
}

function saveData(d) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName(SHEET_NAME);
  if (!sh) {
    sh = ss.insertSheet(SHEET_NAME);
    sh.getRange(1,1).setValue('json');
  }
  sh.getRange(2,1).setValue(JSON.stringify(d));
}

function defaultState() {
  return {
    teams: [], matches: [], photos: [],
    nextMatchId: 1, nextTeamId: 1,
    allowedDomains: [], syncIntervalSec: 15,
    maxPhotoKb: 300, playoffFormat: 'simple'
  };
}
